@startuml
' Sequence Diagram for ServiceTransfer System
skinparam maxMessageSize 150

actor "POS Local Timer\n(otmForm_Timer)" as Timer
participant "wMain.frm" as Form
participant "mMain.bas" as CoreModule
database "Local DB\n(TPSTSalHD, TPSTSalDT...)" as LocalDB
database "Server DB\n(Central Server)" as ServerDB

== Workflow 1: Data Synchronization (Local to Server) ==
Timer -> Form : Trigger Tick Event
activate Form
Form -> CoreModule : SP_CHKbWebservice()
activate CoreModule
CoreModule --> Form : isOnline
deactivate CoreModule

alt isOnline == True
    Form -> LocalDB : SELECT Transactions (FTStaSentOnOff = '0')
    activate LocalDB
    LocalDB --> Form : Transactions Data
    deactivate LocalDB
    
    loop For each Transaction Table (HD, DT, RC, CD, Deposit)
        Form -> ServerDB : INSERT / UPDATE Transaction Data
        activate ServerDB
        ServerDB --> Form : Success
        deactivate ServerDB
        
        Form -> LocalDB : UPDATE FTStaSentOnOff = '1'
    end
end
deactivate Form

== Workflow 2: Token Handling ==
Form -> CoreModule : SP_PRCbToken(CardNo)
activate CoreModule
CoreModule -> ServerDB : SELECT FTToken FROM TPSTTokenLst WHERE FTValue = CardNo
activate ServerDB
ServerDB --> CoreModule : Token Value
deactivate ServerDB

alt Token NOT EXISTS
    CoreModule -> ServerDB : INSERT INTO TPSTTokenLst
end
CoreModule --> Form : Token String
deactivate CoreModule

== Workflow 3: Member Point Update ==
Timer -> Form : W_UPDxUpdatePoint()
activate Form
Form -> LocalDB : SELECT Points FROM TPSTSalePoint WHERE FTRemark = '0'
activate LocalDB
LocalDB --> Form : Member Point Data
deactivate LocalDB

loop For each Member Point Transaction
    Form -> ServerDB : UPDATE TCNMMallCard SET FCEarned = FCEarned + Points
    Form -> ServerDB : INSERT INTO TPSTBPHis (Point History)
    Form -> LocalDB : UPDATE TPSTSalePoint SET FTRemark = '1'
end
deactivate Form

@enduml
